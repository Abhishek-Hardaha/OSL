#!/bin/bash
copy_file()
{
echo "Enter the source file name:"
read src
echo "Enter the destination file path:"
read dest
if cp "$src" "$dest"; then
echo "File copied from $src to $dest"
else
echo "Failed to copy file"
fi
}
move_file()
{
echo "Enter the source file name:"
read src
echo "Enter the destination file path:"
read dest
if mv "$src" "$dest"; then
echo "File moved from $src to $dest"
else
echo "Failed to move file"
fi
}
rename_file()
{
echo "Enter the current file name:"
read src
echo "Enter the new file name:"
read dest
if mv "$src" "$dest"; then
echo "File renamed from $src to $dest"
else
echo "Failed to rename file"
fi
}
delete_file()
{
echo "Enter the file name to delete:"
read file
if rm "$file"; then
echo "File $file deleted"
else
echo "Failed to delete file"
fi
}
while true; do
echo "File Operation Menu:"
echo "1. Copy File"
echo "2. Move File"
echo "3. Rename File"
echo "4. Delete File"
echo "5. Exit"
echo "Enter your choice:"
read choice
case $choice in
1)
copy_file
;;
2)
move_file
;;
3)
rename_file
;;
4)
delete_file
;;
5)
echo "Exiting program."
exit 0
;;
*)
echo "Invalid choice. Please try again."
;;
esac
done
