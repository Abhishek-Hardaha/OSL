#include <stdio.h>
#include <stdlib.h>

#define MAX 100
#define INF 999

int minIndex(int dist[], int visited[], int n) {
    int min = INF, min_index = -1;
    for (int v = 0; v < n; v++) {
        if (!visited[v] && dist[v] < min) {
            min = dist[v];
            min_index = v;
        }
    }
    return min_index;
}

void dijkstra(int graph[MAX][MAX], int src,int V)
{
    int dist[MAX], visited[MAX];

    for (int i = 0; i < V; i++) {
        dist[i] = INF;
        visited[i] = 0;
    }

    dist[src] = 0;


    for (int count = 0; count < V - 1; count++) {
        int u = minIndex(dist, visited, V);
        if (u == -1) break;
        visited[u] = 1;


        for (int v = 0; v < V; v++) {

            if (!visited[v] && graph[u][v] != 0 &&
                dist[u] != INF &&
                dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
                }
        }
    }

    // Print the final shortest distances
    printf("Vertex\tDistance from Source %d\n", src);
    for (int i = 0; i < V; i++)
        printf("%d - > %d\n", i, dist[i]);

}

int main()
{
    int n;
    int graph[MAX][MAX];
    int src;
    printf("---------- Dijkstra's Algorithm ----------\n");

    printf("Enter number of nodes: ");
    scanf("%d", &n);

    printf("(Graph nodes start from 0 to %d): \n Enter the Source -> ", n-1);
    scanf("%d", &src);


    printf("Enter adjacency matrix: \n");

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {

            scanf("%d", &graph[i][j]);

        }
    }

    dijkstra(graph,src,n);
}
