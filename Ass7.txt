#include <stdio.h>
float calculate_average(int arr[], int n) {
int sum = 0;
for (int i=0; i < n; i++) {
sum += arr[i];
}
return (float)sum / n;
}
int find_maximum(int arr[], int n) {
int max = arr[0];
for (int i=1; i < n; i++) {
if (arr[i] > max) {
max = arr[i];
}
}
return max;
}
void input_array(int arr[], int n) {
for (int i=0; i < n; i++) {
printf("Element %d: ", i + 1);
scanf("%d", &arr[i]);
}
}
int main() {
int n;
printf("Enter the number of elements in the array: ");
scanf("%d", &n);
int arr[n];
printf("Enter %d elements: \n", n);
input_array(arr, n);
float avg = calculate_average(arr, n);
int max = find_maximum(arr, n);
printf("\nAverage of the array: %.2f\n", avg);
printf("Maximum element in the array: %d\n", max);
return 0;
}
